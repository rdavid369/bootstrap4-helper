def initialize(template, opts = {}, &block)
  super(template)

  @id      = opts.fetch(:id,    '')
  @class   = opts.fetch(:class, '')
  @data    = opts.fetch(:data,  nil)
  @content = block || proc { '' }
end

def to_s
  content_tag :div, id: @id, class: "card #{@class}", data: @data do
    @content.call(self)
  end
end
